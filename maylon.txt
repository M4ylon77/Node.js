app.post('/users', (req, res) => {
  const { name, age, occupation } = req.body;

  const newUser = new User({ name, age, occupation });
  newUser.save()
    .then(() => {
      res.redirect('/');
    })
    .catch((error) => {
      console.error('Erro ao salvar o usuário:', error);
      res.status(500).send('Erro interno do servidor');
    });
});
